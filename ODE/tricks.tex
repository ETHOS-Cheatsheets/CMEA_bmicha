\subsubsection{Non-Autonomous \texorpdfstring{$\rightarrow$}{->} Autonomous}
    \vspace{2pt}
    \mathbox{
        u'(t) = F(t,u(t)) \quad \rightarrow \quad w'(t) = G(w(t))
    }
    $$
        w(t) = [u^T(t), \alpha(t)]^T, \quad w(0) = [u^T(0), \alpha(0)]^T
    $$
    $$
        G(w) = w'(t) = [u'(t)^T, \alpha'(t)]^T
    $$
    \vspace{-1em}
